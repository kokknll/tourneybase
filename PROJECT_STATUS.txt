========================================
  PROJECT STRUCTURE
========================================

src/
  lib/
    supabase/
      client.ts      — Supabase browser client
      server.ts      — Supabase server client (for Server Components/Actions)
      middleware.ts   — Supabase middleware session helper
  components/
    Navbar.tsx       — Navbar with auth state (login/signup or user/logout)
  app/
    globals.css      — Global styles, CSS variables, dark theme
    layout.tsx       — Root layout, fonts, metadata
    page.tsx         — Landing page (hero, features, CTA, footer)
    login/
      page.tsx       — Login page (email/password)
    signup/
      page.tsx       — Signup page (email/password + confirmation)
    auth/
      callback/
        route.ts     — Auth callback handler (email confirmation)
  middleware.ts      — Next.js middleware (session refresh on all routes)
